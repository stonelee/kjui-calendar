有初始值

<input id="date1" type="text">
<input id="date2" type="text" value="2013-02-12">

<script>
seajs.use(['$', 'kjui/calendar/1.0.0/calendar'], function($, Calendar) {
  new Calendar({
    target: '#date1',
    focus: '2013-1-1'
  });

  new Calendar({
    target: '#date2'
  });
});
</script>

控制可选范围

<div class="group">
  <label>2013-5-7至2013-5-21:</label>
  <input id="date1" type="text">
</div>
<div class="group">
  <label>2013-5-15之后:</label>
  <input id="date2" type="text">
</div>
<div class="group">
  <label>不能选周末:</label>
  <input id="date3" type="text">
</div>

<script>
seajs.use(['$', 'kjui/calendar/1.0.0/calendar'], function($, Calendar) {
  new Calendar({
    target: '#date1',
    range: ['2013-05-07', '2013-05-21']
  });

  new Calendar({
    target: '#date2',
    range: ['2013-05-15', null]
  });

  new Calendar({
    target: '#date3',
    range: function(date) {
      return date.day() != 0 && date.day() != 6;
    }
  });
});
</script>
